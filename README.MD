'''
node --version

# To the initialize the firebase console

'''
firebase init
'''

'''
npm run server
'''

# To create react project usin npm & yarn

'''
npm create-react-app folder-name
'''

'''
yarn create-react-app folder-name
'''

# To start the client

'''
yarn start
'''
